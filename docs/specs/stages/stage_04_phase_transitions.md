def calculate_transition_enthalpy(
    solid_record: DatabaseRecord,
    liquid_record: DatabaseRecord,
    T_melt: float

    """

    Returns:
        (ŒîH_fusion –≤ –∫–î–∂/–º–æ–ª—å, –º–µ—Ç–æ–¥: 'calculated' –∏–ª–∏ 'heuristic')
    """
    # –®–∞–≥ 1: H –Ω–∞ T_melt –¥–ª—è —Ç–≤—ë—Ä–¥–æ–π —Ñ–∞–∑—ã
    H_solid_at_melt = calculate_enthalpy_at_temp(solid_record, T_melt)
    
    # –®–∞–≥ 2: H –Ω–∞ T_melt –¥–ª—è –∂–∏–¥–∫–æ–π —Ñ–∞–∑—ã
    H_liquid_at_melt = calculate_enthalpy_at_temp(liquid_record, T_melt)
    
    # –®–∞–≥ 3: –†–∞–∑–Ω–∏—Ü–∞ –¥–∞—ë—Ç –ø—Ä–∏–±–ª–∏–∂—ë–Ω–Ω—É—é —ç–Ω—Ç–∞–ª—å–ø–∏—é –ø–ª–∞–≤–ª–µ–Ω–∏—è

    else:

        # –ò—Å–ø–æ–ª—å–∑—É–µ–º —ç–≤—Ä–∏—Å—Ç–∏–∫—É
        logger.warning(
            f"–†–∞—Å—á—ë—Ç–Ω–∞—è ŒîH_fusion = {delta_H_fusion:.2f} < 0. "
            f"–ò—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è —ç–≤—Ä–∏—Å—Ç–∏—á–µ—Å–∫–∞—è –æ—Ü–µ–Ω–∫–∞."

def estimate_fusion_enthalpy(record: DatabaseRecord, T_melt: float) -> float:
    """

    """
    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–∏–ø –≤–µ—â–µ—Å—Ç–≤–∞ –ø–æ —Ñ–æ—Ä–º—É–ª–µ
    compound_type = classify_compound(record.formula)
    
    # –¢–∏–ø–∏—á–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è ŒîS_fusion (–î–∂/(–º–æ–ª—å¬∑K)):
    ds_fusion_estimates = {
        'metal': 10.0,      # –ú–µ—Ç–∞–ª–ª—ã
        'salt': 25.0,       # –°–æ–ª–∏
        'molecular': 15.0,  # –ú–æ–ª–µ–∫—É–ª—è—Ä–Ω—ã–µ

    delta_h_fusion = T_melt * ds_fusion / 1000  # ‚Üí –∫–î–∂/–º–æ–ª—å
    
    return delta_h_fusion
